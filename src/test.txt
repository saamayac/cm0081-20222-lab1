-- Variables.
x;

myVariable42;

-- An application.
x y;

-- \-abstraction.
\x.x;

-- Outermost parentheses are not written.
\x.x;

-- Application has higher precedence.
\x.x y;

-- Application associates to the left.
x y z;

-- Abstraction associates to the right.
\x.\y.\z.x y z;

-- The Î© term.
(\x.x x)(\x.x x);
